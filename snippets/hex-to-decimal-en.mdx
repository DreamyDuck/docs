export const HexToDecimalConverter = () => {
  const [hexInput, setHexInput] = useState('')
  const [decimalResult, setDecimalResult] = useState('')
  const [isValid, setIsValid] = useState(true)
  const [copied, setCopied] = useState(false)

  const validateAndConvertHex = (hexValue) => {
    // Remove # if present and trim whitespace
    const cleanHex = hexValue.replace('#', '').trim()
    
    // Check if it's a valid hex string (only contains 0-9, A-F, a-f)
    const hexRegex = /^[0-9A-Fa-f]+$/
    
    if (cleanHex === '') {
      setDecimalResult('')
      setIsValid(true)
      return
    }
    
    if (!hexRegex.test(cleanHex)) {
      setIsValid(false)
      setDecimalResult('')
      return
    }
    
    try {
      // Convert hex to decimal using parseInt with base 16
      const decimal = parseInt(cleanHex, 16)
      setDecimalResult(decimal.toString())
      setIsValid(true)
    } catch (error) {
      setIsValid(false)
      setDecimalResult('')
    }
  }

  const handleInputChange = (e) => {
    const value = e.target.value
    setHexInput(value)
    validateAndConvertHex(value)
    setCopied(false)
  }

  const handleCopyToClipboard = async () => {
    if (decimalResult) {
      try {
        await navigator.clipboard.writeText(decimalResult)
        setCopied(true)
        setTimeout(() => setCopied(false), 2000)
      } catch (err) {
        console.error('Failed to copy: ', err)
      }
    }
  }

  const handleKeyDown = (e) => {
    if (e.key === 'Enter' && decimalResult) {
      handleCopyToClipboard()
    }
  }

  const examples = [
    { hex: 'FF', decimal: '255' },
    { hex: '1A', decimal: '26' },
    { hex: 'ABC', decimal: '2748' },
    { hex: '1234', decimal: '4660' }
  ]

  const handleExampleClick = (hexValue) => {
    setHexInput(hexValue)
    validateAndConvertHex(hexValue)
    setCopied(false)
  }

  return (
    <div className="p-6 border dark:border-zinc-950/80 rounded-xl not-prose">
      <div className="space-y-6">
        <div className="space-y-3">
          <label htmlFor="hex-input" className="block text-sm font-medium text-zinc-950/70 dark:text-white/70">
            Hexadecimal Input
          </label>
          <div className="relative">
            <span className="absolute left-3 top-1/2 transform -translate-y-1/2 text-zinc-950/50 dark:text-white/50">
              #
            </span>
            <input
              id="hex-input"
              type="text"
              value={hexInput}
              onChange={handleInputChange}
              onKeyDown={handleKeyDown}
              placeholder="Enter hex value (e.g., FF, 1A, ABC)"
              className={`w-full pl-8 pr-3 py-3 rounded-lg border text-lg font-mono transition-colors ${
                !isValid 
                  ? 'border-red-500 bg-red-50 dark:bg-red-950/20 dark:border-red-500' 
                  : 'border-zinc-300 dark:border-zinc-600 bg-white dark:bg-zinc-800 focus:border-blue-500 focus:ring-2 focus:ring-blue-500/20'
              }`}
              aria-label="Hexadecimal input field"
              tabIndex={0}
            />
          </div>
          {!isValid && (
            <p className="text-sm text-red-600 dark:text-red-400">
              Please enter a valid hexadecimal value (0-9, A-F)
            </p>
          )}
        </div>

        {decimalResult && (
          <div className="space-y-3">
            <label className="block text-sm font-medium text-zinc-950/70 dark:text-white/70">
              Decimal Result
            </label>
            <div className="flex items-center space-x-3">
              <div className="flex-1 p-3 bg-zinc-100 dark:bg-zinc-800 rounded-lg">
                <span className="text-2xl font-mono text-zinc-950 dark:text-white">
                  {decimalResult}
                </span>
              </div>
              <button
                onClick={handleCopyToClipboard}
                onKeyDown={(e) => e.key === 'Enter' && handleCopyToClipboard()}
                className={`px-4 py-3 rounded-lg font-medium transition-colors ${
                  copied
                    ? 'bg-green-500 text-white'
                    : 'bg-blue-500 hover:bg-blue-600 text-white'
                }`}
                aria-label={copied ? 'Copied to clipboard' : 'Copy decimal result to clipboard'}
                tabIndex={0}
              >
                {copied ? 'Copied!' : 'Copy'}
              </button>
            </div>
          </div>
        )}

        <div className="space-y-3">
          <h3 className="text-sm font-medium text-zinc-950/70 dark:text-white/70">
            Quick Examples
          </h3>
          <div className="grid grid-cols-2 gap-2">
            {examples.map((example, idx) => (
              <button
                key={idx}
                onClick={() => handleExampleClick(example.hex)}
                onKeyDown={(e) => e.key === 'Enter' && handleExampleClick(example.hex)}
                className="p-3 text-left rounded-lg border border-zinc-300 dark:border-zinc-600 hover:bg-zinc-50 dark:hover:bg-zinc-800 transition-colors"
                aria-label={`Example: ${example.hex} converts to ${example.decimal}`}
                tabIndex={0}
              >
                <div className="text-sm font-mono text-blue-600 dark:text-blue-400">
                  #{example.hex}
                </div>
                <div className="text-sm text-zinc-950/70 dark:text-white/70">
                  = {example.decimal}
                </div>
              </button>
            ))}
          </div>
        </div>

        <div className="text-xs text-zinc-950/50 dark:text-white/50">
          <p>Tip: You can include or omit the # prefix. Press Enter to copy the result.</p>
        </div>
      </div>
    </div>
  )
}
