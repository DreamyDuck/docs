export const HexDecimalConverter = () => {
  const [hexInput, setHexInput] = useState('FF')
  const [decimalResult, setDecimalResult] = useState(255)
  const [isValid, setIsValid] = useState(true)
  const [conversions, setConversions] = useState([])

  useEffect(() => {
    // Remove any non-hex characters and convert to uppercase
    const cleanHex = hexInput.replace(/[^0-9A-Fa-f]/g, '').toUpperCase()
    
    if (cleanHex === '') {
      setDecimalResult(0)
      setIsValid(true)
      setConversions([])
      return
    }

    try {
      const decimal = parseInt(cleanHex, 16)
      setDecimalResult(decimal)
      setIsValid(true)
      
      // Generate related conversions
      const newConversions = []
      const baseValue = decimal
      
      // Generate some related hex values
      for (let i = -2; i <= 2; i++) {
        const value = Math.max(0, baseValue + i)
        const hex = value.toString(16).toUpperCase()
        if (hex !== cleanHex) {
          newConversions.push({ hex, decimal: value })
        }
      }
      
      setConversions(newConversions.slice(0, 5))
    } catch (error) {
      setIsValid(false)
      setDecimalResult(0)
      setConversions([])
    }
  }, [hexInput])

  const copyToClipboard = (value) => {
    navigator.clipboard
      .writeText(value.toString())
      .then(() => {
        console.log(`Copied ${value} to clipboard!`)
      })
      .catch((err) => {
        console.error("Failed to copy: ", err)
      })
  }

  const handleInputChange = (e) => {
    const value = e.target.value
    setHexInput(value)
  }

  const presetHexValues = ['FF', '100', '1A2B', 'CAFE', 'DEAD', 'BEEF']

  return (
    <div className="p-4 border dark:border-zinc-950/80 rounded-xl not-prose">
      <div className="space-y-4">
        <div className="space-y-2">
          <label className="block text-sm text-zinc-950/70 dark:text-white/70">
            Hexadecimal Input
            <input
              type="text"
              value={hexInput}
              onChange={handleInputChange}
              placeholder="Enter hex value (e.g., FF, 1A2B)"
              className={`w-full p-2 mt-1 border rounded-lg font-mono text-lg ${
                isValid 
                  ? 'border-zinc-300 dark:border-zinc-600' 
                  : 'border-red-500'
              } bg-white dark:bg-zinc-800 text-zinc-900 dark:text-white`}
            />
          </label>
          
          {!isValid && (
            <p className="text-red-500 text-sm">
              Invalid hexadecimal input. Use only 0-9 and A-F.
            </p>
          )}
        </div>

        <div className="bg-zinc-50 dark:bg-zinc-800/50 p-4 rounded-lg">
          <div className="text-center">
            <div className="text-sm text-zinc-950/70 dark:text-white/70 mb-1">
              Decimal Result
            </div>
            <div className="text-3xl font-mono font-bold text-zinc-900 dark:text-white">
              {decimalResult.toLocaleString()}
            </div>
            <button
              onClick={() => copyToClipboard(decimalResult)}
              className="mt-2 px-3 py-1 text-xs bg-blue-500 text-white rounded hover:bg-blue-600 transition-colors"
            >
              Copy Decimal
            </button>
          </div>
        </div>

        <div className="space-y-2">
          <div className="text-sm text-zinc-950/70 dark:text-white/70">
            Quick Presets
          </div>
          <div className="flex flex-wrap gap-2">
            {presetHexValues.map((preset) => (
              <button
                key={preset}
                onClick={() => setHexInput(preset)}
                className="px-3 py-1 text-sm font-mono bg-zinc-200 dark:bg-zinc-700 text-zinc-800 dark:text-zinc-200 rounded hover:bg-zinc-300 dark:hover:bg-zinc-600 transition-colors"
              >
                {preset}
              </button>
            ))}
          </div>
        </div>

        {conversions.length > 0 && (
          <div className="space-y-2">
            <div className="text-sm text-zinc-950/70 dark:text-white/70">
              Related Conversions
            </div>
            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-2">
              {conversions.map((conversion, idx) => (
                <div
                  key={idx}
                  className="p-2 bg-zinc-100 dark:bg-zinc-800 rounded cursor-pointer hover:bg-zinc-200 dark:hover:bg-zinc-700 transition-colors"
                  onClick={() => copyToClipboard(conversion.decimal)}
                  title={`Click to copy decimal: ${conversion.decimal}`}
                >
                  <div className="text-xs text-zinc-600 dark:text-zinc-400">
                    {conversion.hex}₁₆
                  </div>
                  <div className="font-mono font-bold text-zinc-900 dark:text-white">
                    {conversion.decimal.toLocaleString()}₁₀
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        <div className="text-sm font-mono text-zinc-950/70 dark:text-white/70 bg-zinc-50 dark:bg-zinc-800/30 p-3 rounded">
          <div className="grid grid-cols-1 sm:grid-cols-2 gap-2">
            <div>
              <strong>Hex:</strong> {hexInput || '0'}₁₆
            </div>
            <div>
              <strong>Decimal:</strong> {decimalResult.toLocaleString()}₁₀
            </div>
          </div>
        </div>
      </div>
    </div>
  )
}
